```typescript
// src/session/registry.ts — full interface to implement

export type ChatSessions = {
  default: string;              // auto-created sessionId
  named: Map<string, string>;   // name (lowercase) → sessionId
  active: string;               // currently selected sessionId
};

export class SessionRegistry {
  // Returns active sessionId or undefined if chat has no sessions yet
  getActiveSessionId(chatId: number): string | undefined;

  // Returns the display name for the active session ("default" or named key)
  // Returns "default" if no sessions exist yet
  getActiveName(chatId: number): string;

  // Creates default session via createSession(openCodeUrl) if not yet created.
  // Sets active to default. Returns active sessionId.
  async getOrCreateDefault(chatId: number, openCodeUrl: string): Promise<string>;

  // Registers a pre-created named session (caller must already have called createSession)
  // Sets active to the new named session
  createNamed(chatId: number, name: string, sessionId: string): void;

  // Switches active to named session. Returns true if found, false if not found.
  switchTo(chatId: number, name: string): boolean;

  // Returns true if named session exists (after lowercase normalization)
  hasNamed(chatId: number, name: string): boolean;

  // Returns sessionId for a named session or undefined
  getNamedId(chatId: number, name: string): string | undefined;

  // Lists all sessions for a chat: [{name, sessionId, active}]
  // Returns [] if chat has no sessions yet (never throws)
  list(chatId: number): Array<{ name: string; sessionId: string; active: boolean }>;
}
```
