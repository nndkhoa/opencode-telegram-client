**src/config/env.ts** — Export a `parseEnv(raw: NodeJS.ProcessEnv)` function (testable, takes env object) AND a top-level `config` singleton that calls `parseEnv(process.env)` and calls `process.exit(1)` with a clear log on failure (per D-05):

```typescript
import "dotenv/config";
import { z } from "zod";
import { logger } from "../logger.js";

const EnvSchema = z.object({
  BOT_TOKEN: z.string().min(1, "BOT_TOKEN is required"),
  OPENCODE_URL: z.string().url().default("http://localhost:4096"),
  ALLOWED_USER_IDS: z
    .string()
    .min(1, "ALLOWED_USER_IDS must be a non-empty comma-separated list of Telegram user IDs")
    .transform((s) =>
      s
        .split(",")
        .map((x) => x.trim())
        .filter(Boolean)
        .map((x) => {
          const n = Number.parseInt(x, 10);
          if (Number.isNaN(n) || n <= 0) throw new Error(`Invalid user ID: "${x}"`);
          return n;
        })
    )
    .pipe(z.array(z.number().int().positive()).nonempty()),
});

export type Config = {
  botToken: string;
  openCodeUrl: string;
  allowedUserIds: Set<number>;
};

export function parseEnv(raw: NodeJS.ProcessEnv): Config {
  const result = EnvSchema.safeParse(raw);
  if (!result.success) {
    throw new Error(result.error.issues.map((i) => i.message).join("; "));
  }
  const d = result.data;
  return {
    botToken: d.BOT_TOKEN,
    openCodeUrl: d.OPENCODE_URL,
    allowedUserIds: new Set(d.ALLOWED_USER_IDS),
  };
}

let _config: Config;
try {
  _config = parseEnv(process.env);
} catch (err) {
  logger.fatal({ err }, "Invalid configuration — fix env vars and restart");
  process.exit(1);
}
export const config = _config;
```

Note: `import "dotenv/config"` at the top ensures `.env` is loaded before Zod runs when this module is imported in production. Tests pass raw objects directly to `parseEnv()` and do NOT import the module top-level (to avoid process.exit in test context).

```typescript
import { describe, it, expect } from "vitest";
import { parseEnv } from "./env.js";

describe("parseEnv", () => {
  const valid = {
    BOT_TOKEN: "abc123",
    ALLOWED_USER_IDS: "123456,789012",
    OPENCODE_URL: "http://localhost:4096",
  };

  it("returns typed config from valid env", () => {
    const cfg = parseEnv(valid);
    expect(cfg.botToken).toBe("abc123");
    expect(cfg.openCodeUrl).toBe("http://localhost:4096");
    expect(cfg.allowedUserIds).toEqual(new Set([123456, 789012]));
  });

  it("defaults OPENCODE_URL to http://localhost:4096", () => {
    const cfg = parseEnv({ BOT_TOKEN: "tok", ALLOWED_USER_IDS: "1" });
    expect(cfg.openCodeUrl).toBe("http://localhost:4096");
  });

  it("throws when BOT_TOKEN is missing", () => {
    expect(() => parseEnv({ ALLOWED_USER_IDS: "1" })).toThrow();
  });

  it("throws when ALLOWED_USER_IDS is empty string", () => {
    expect(() => parseEnv({ BOT_TOKEN: "tok", ALLOWED_USER_IDS: "" })).toThrow();
  });

  it("throws when ALLOWED_USER_IDS contains non-numeric value", () => {
    expect(() => parseEnv({ BOT_TOKEN: "tok", ALLOWED_USER_IDS: "abc" })).toThrow();
  });

  it("parses comma-separated user IDs into Set", () => {
    const cfg = parseEnv({ BOT_TOKEN: "tok", ALLOWED_USER_IDS: "100,200,300" });
    expect(cfg.allowedUserIds).toEqual(new Set([100, 200, 300]));
  });
});
```
