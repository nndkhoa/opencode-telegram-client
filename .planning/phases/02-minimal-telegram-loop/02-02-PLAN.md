```typescript
export async function createSession(baseUrl: string): Promise<string> {
  const res = await fetch(new URL("/session", baseUrl).toString(), {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({}),
  });
  if (!res.ok) throw new Error(`POST /session failed: HTTP ${res.status}`);
  const data = (await res.json()) as { id: string };
  return data.id;
}

export async function sendPromptAsync(
  baseUrl: string,
  sessionId: string,
  text: string
): Promise<void> {
  const url = new URL(`/session/${sessionId}/prompt_async`, baseUrl).toString();
  const res = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      parts: [{ type: "text", text }],
    }),
  });
  if (!res.ok && res.status !== 204) {
    throw new Error(`prompt_async failed: HTTP ${res.status}`);
  }
}
```

Key implementation notes:
- `POST /session` body is `{}` (empty JSON object) — verified against live OpenCode 1.3.3
- `prompt_async` returns 204 No Content — the check is `!res.ok && res.status !== 204` (204 is falsy for `ok` in some environments but represents success)
- Use `.js` extension on any future imports within this file (ESM/NodeNext convention)
- No logger import needed — errors are thrown; caller handles logging

```typescript
import type { Api } from "grammy";
import type { OpenCodeEvent } from "./events.js";

const THROTTLE_MS = 500;

type TurnState = {
  chatId: number;
  messageId: number;
  buffer: string;
  lastEditAt: number;
};

export class StreamingStateManager {
  // chatId → sessionId (created on first message per D-01)
  private sessions = new Map<number, string>();
  // chatId → busy flag (D-08)
  private busy = new Map<number, boolean>();
  // sessionId → active turn state
  // Exposed as non-private for test access via type assertion
  turns = new Map<string, TurnState>();

  getSession(chatId: number): string | undefined {
    return this.sessions.get(chatId);
  }

  setSession(chatId: number, sessionId: string): void {
    this.sessions.set(chatId, sessionId);
  }

  isBusy(chatId: number): boolean {
    return this.busy.get(chatId) === true;
  }

  startTurn(sessionId: string, chatId: number, messageId: number): void {
    this.busy.set(chatId, true);
    this.turns.set(sessionId, {
      chatId,
      messageId,
      buffer: "",
      lastEditAt: Date.now(),
    });
  }

  endTurn(sessionId: string): void {
    const turn = this.turns.get(sessionId);
    if (turn) {
      this.busy.set(turn.chatId, false);
      this.turns.delete(sessionId);
    }
  }

  handleEvent(event: OpenCodeEvent, bot: Api): void {
    if (event.type === "message.part.delta") {
      const { sessionID, field, delta } = event.properties;
      // Filter: only accumulate text-field deltas (per research open question 3)
      if (field !== "text" || !delta) return;

      const turn = this.turns.get(sessionID);
      if (!turn) return;

      turn.buffer += delta;

      // D-05: throttle editMessageText to ~500ms
      const now = Date.now();
      if (now - turn.lastEditAt >= THROTTLE_MS) {
        turn.lastEditAt = now;
        const interim = `⏳ Thinking...\n\n${turn.buffer}`;
        bot
          .editMessageText(turn.chatId, turn.messageId, interim)
          .catch(() => {
            // Silently ignore "message is not modified" Telegram 400 errors
          });
      }
      return;
    }

    if (event.type === "session.idle") {
      const { sessionID } = event.properties;
      const turn = this.turns.get(sessionID);
      if (!turn) return;

      // D-04: capture state, clear turn FIRST (prevents race with throttled edits per Pitfall 5)
      const finalText = turn.buffer || "(empty response)";
      const { chatId, messageId } = turn;
      this.endTurn(sessionID); // clears busy flag + removes from turns map

      // D-02: session.idle is the definitive turn-end signal — send final clean message
      bot
        .editMessageText(chatId, messageId, finalText)
        .catch(() => {});
    }
  }
}
```

Critical implementation notes:
- `turns` map is NOT private — test file accesses it via type assertion `(manager as unknown as { turns: Map<... }}).turns`
- `endTurn` called BEFORE `editMessageText` on `session.idle` (Pitfall 5 — race condition prevention)
- `field !== "text"` filter prevents tool call deltas from polluting the buffer (open question 3 from research)
- All `editMessageText` calls have `.catch(() => {})` to swallow "message not modified" Telegram errors (Pitfall 2)
- `.js` extension on all imports (ESM/NodeNext convention)
